for ir:

#define IR_PIN 4

void setup() {
  Serial.begin(115200);
  pinMode(IR_PIN, INPUT);
}

void loop() {
  int irValue = digitalRead(IR_PIN);

  if (irValue == 0) {
    Serial.println("Helmet WORN");
  } else {
    Serial.println("Helmet NOT WORN");
  }

  delay(500);
}

for mpu6050:

#include <Wire.h>
#include <MPU6050.h>

MPU6050 mpu;

void setup() {
  Serial.begin(115200);
  Wire.begin();
  mpu.initialize();

  if (mpu.testConnection()) {
    Serial.println("MPU6050 Connected");
  } else {
    Serial.println("MPU6050 Failed");
  }
}

void loop() {
  int16_t ax, ay, az;
  mpu.getAcceleration(&ax, &ay, &az);

  Serial.print("Accel X: "); Serial.print(ax);
  Serial.print(" Y: "); Serial.print(ay);
  Serial.print(" Z: "); Serial.println(az);

  delay(500);
}


for fsr:

const int fsrPin = 34;     // Analog pin
const float VCC = 3.3;     // ESP32 voltage
const int ADC_RES = 4095;  // 12-bit ADC

void setup() {
  Serial.begin(115200);
}

void loop() {
  int fsrValue = analogRead(fsrPin);

  // Convert to voltage
  float voltage = (fsrValue * VCC) / ADC_RES;

  Serial.print("Raw ADC: ");
  Serial.print(fsrValue);

  Serial.print(" | Voltage: ");
  Serial.print(voltage);
  Serial.println(" V");

  delay(500);
}


const int fsrPin = 34;

void setup() {
  Serial.begin(115200);
}

void loop() {
  int fsrValue = analogRead(fsrPin);

  Serial.print("FSR Value: ");
  Serial.print(fsrValue);

  if (fsrValue < 200) {
    Serial.println("  -> No Pressure");
  }
  else if (fsrValue < 1500) {
    Serial.println("  -> Light Pressure");
  }
  else if (fsrValue < 3000) {
    Serial.println("  -> Medium Pressure");
  }
  else {
    Serial.println("  -> High Pressure");
  }

  delay(300);
}



for fsr blinking:
const int fsrPin = 34;
const int ledPin = 2;
const int threshold = 1000;

unsigned long previousMillis = 0;
const long interval = 200;
bool ledState = false;

void setup() {
  pinMode(ledPin, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  int fsrValue = analogRead(fsrPin);

  if (fsrValue > threshold) {
    unsigned long currentMillis = millis();
    if (currentMillis - previousMillis >= interval) {
      previousMillis = currentMillis;
      ledState = !ledState;
      digitalWrite(ledPin, ledState);
    }
  } 
  else {
    digitalWrite(ledPin, LOW);
  }
}




const int irPin = 27;     // IR sensor output
const int ledPin = 2;     // LED pin

void setup() {
  pinMode(irPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Serial.begin(115200);
}

void loop() {
  int irState = digitalRead(irPin);

  // Most IR sensors give LOW when object detected
  if (irState == LOW) {
    digitalWrite(ledPin, HIGH);
    delay(200);
    digitalWrite(ledPin, LOW);
    delay(200);
  } 
  else {
    digitalWrite(ledPin, LOW);
  }
}
